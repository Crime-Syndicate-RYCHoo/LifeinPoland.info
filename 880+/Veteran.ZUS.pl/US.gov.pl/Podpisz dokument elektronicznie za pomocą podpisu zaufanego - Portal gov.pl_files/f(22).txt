jQuery('#signer').on('click', '.drop-file-upload__remove-file', function () {
    var path = sessionStorage.getItem('path').toString();
    window.piwikActions.sendPiwikAction('ikona', 'usuń dodany dokument');
    window.piwikActions.updateDocTitle("Podpisz dokument elektronicznie za pomocą podpisu zaufanego " + path + " - Portal gov.pl");
});
